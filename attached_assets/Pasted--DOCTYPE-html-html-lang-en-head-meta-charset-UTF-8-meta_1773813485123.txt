    // Function: Navigation Logic
    function navigateTo(pageId) {
        const pages = ['home', 'curriculum', 'vault'];
        pages.forEach(id => {
            const el = document.getElementById(id);
            if (id === pageId) {
                el.classList.remove('hidden-page');
                el.classList.add('active-page');
            } else {
                el.classList.add('hidden-page');
                el.classList.remove('active-page');
            }
        });
        window.scrollTo({ top: 0, behavior: 'smooth' });
    }
    // Initialize: Build Module List
    const moduleListContainer = document.getElementById('module-list');
    modules.forEach((module, index) => {
        const num = (index + 1).toString().padStart(2, '0');
        const item = document.createElement('div');
        item.className = "module-row flex items-center gap-4 py-4 border-b border-black/[0.03] transition-all duration-300 cursor-default";
        item.innerHTML = `
            <span class="text-[10px] font-bold opacity-20 tracking-tighter">${num}</span>
            <span class="text-[10px] uppercase tracking-widest font-medium opacity-80">${module}</span>
        `;
        moduleListContainer.appendChild(item);
    });
    // Function: Accordion Toggle
    function toggleAccordion(btn) {
        const item = btn.parentElement;
        const isActive = item.classList.contains('active');
        document.querySelectorAll('.accordion-item').forEach(el => el.classList.remove('active'));
        if (!isActive) item.classList.add('active');
    }
